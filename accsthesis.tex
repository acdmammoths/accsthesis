% A template for typesetting an honors thesis in the Computer Science Department
% at Amherst College.
%
% By design, it is very minimal and only includes packages that are used for the
% formatting.
%
\documentclass[11pt,twoside,openright]{report}
% 11pt: Set 11pt font size
% twoside,openright: start chapters on right-hand pages (openright requires
% twoside. We undo the effects of twoside by using the geometry package.)

%% Required packages: do not remove or change %%
\usepackage[margin=1in]{geometry} % Set all margins to 1 inch. This setting
% also undoes the effects of the twoside class option, and it is fine because
% theses are rarely bound these days, which is what twoside is really for.
%
% Use nice fonts, inspired by the ACM 'acmart' class.
\usepackage{libertinus} % libertinus for text
\usepackage{inconsolata} % inconsolata for math
%
% Package used to set double spacing
\usepackage{setspace}
%% End of required packages %%

%% Fill the following four lines: it's all you need to do
\author{Student Name}
\title{Thesis Title}
\newcommand{\classyear}{2021} % Your class year
\newcommand{\advisor}{Advisor Name} % "Prof." will be added by the style.

\usepackage{lipsum} % Remove this package when you start writing

\begin{document} % Let's go!

%% The title page: do not remove or change %%
\begin{titlepage}
  % This title page is in part inspired by the titleDS example in the titlepages
  % LaTeX package
  \centering
  {\LARGE\makeatletter\textbf{\MakeUppercase\@title}\makeatother}\par
  \vspace{0.6\baselineskip}
  %{By}\\[0.6\baselineskip]
  {\scshape%\makeatletter\@author\makeatother\\[0.6\baselineskip]
    Submitted to the %\\[0.5\baselineskip]
    Department of Computer Science %\\[0.5\baselineskip]
    of Amherst College \\[0.5\baselineskip]
    in partial fulfillment of the requirements \\[0.5\baselineskip]
    for the degree of %\\[0.5\baselineskip]
    Bachelor of Arts with honors
  }\par
  \vspace{0.6\baselineskip}
  \textsc{\today}\\[1in] % we just use \today for the date
  \vspace{8\baselineskip}
  {\textsc{Author:} \makeatletter\@author\makeatother%
  \hfill\textsc{Advisor:} Prof.\ \advisor}\\[0.6\baselineskip]
  \vfill
  \textsc{Copyright} \copyright\ \textsc{\classyear}\ % the space is important.
  {\scshape \makeatletter\@author\makeatother}
\end{titlepage}

\pagenumbering{roman} % Use Roman numbers for the frontmatter pages

\chapter*{Abstract} % An abstract is required

Your abstract here (1 page at most)

\chapter*{Acknowledgements} % Acknowledgements are optional

Thanks, John von Neumann!

\tableofcontents % Required table of contents after the abstract and acknowledgements

\doublespacing% Use double spacing in the main text

\chapter{Your First Chapter Title in Title Case}
\pagenumbering{arabic} % Use Arab numbers for the main body pages. This command
                       % must be here otherwise the ToC gets arabic page numbers

\lipsum% Remove this when you start writing.

\bibliographystyle{plain} % Use the ``plain'' bibliography style. If you are
% fancy and want to use the natbib package (you should), remember to change this
% to ``plainnat''.

%\bibliography{thesisRefs} % Replace with your BibTeX file name and uncomment.

\end{document}
